function orderFromChinese() { const inputText = document.getElementById("orderInput").value.trim(); if (!inputText) { alert("請輸入訂單內容！"); return; } // 轉譯中文為對應字母，並顯示轉譯結果 const translatedText = translateDishName(inputText); document.getElementById("convertedText").innerText = "轉譯結果：" + translatedText; // 將 "/" 替換成空白，再以空白分割 token let tokens = translatedText.replace(/\//g, " ").split(/\s+/).filter(t => t !== ""); // 分離純數字 token 與餐點 token let dishOrders = []; let numericTokens = []; tokens.forEach(token => { if (/^\d+$/.test(token)) { // 純數字 token numericTokens.push(parseInt(token)); } else { // 餐點 token，格式須為字母或字母＋數字 let m = token.match(/^([a-zA-Z]+)(\d+)?$/); if (!m) { alert("訂單格式錯誤：" + token); return; } dishOrders.push({ keyword: m[1], quantity: m[2] ? parseInt(m[2]) : null }); } }); // 限制一次最多只能輸入兩組數字與兩個餐點 if (numericTokens.length > 2) { alert("一次最多只能輸入兩組數字！"); return; } if (dishOrders.length > 2) { alert("一次最多只能輸入兩個餐點！"); return; } // 依照數字匹配規則為餐點補上數量： // 若只有一組數字，則缺數量者均採用該數字； // 若有兩組數字，則依序匹配（第一個餐點補第一組、第二個餐點補第二組）。 if (numericTokens.length === 1) { dishOrders.forEach(order => { if (order.quantity === null) { order.quantity = numericTokens[0]; } }); } else if (numericTokens.length === 2) { // 依照餐點順序檢查是否已附數量，若未附則依序分配 for (let i = 0; i < dishOrders.length; i++) { if (dishOrders[i].quantity === null && i < numericTokens.length) { dishOrders[i].quantity = numericTokens[i]; } } } // 若仍有未設定數量者，預設為 1 份 dishOrders.forEach(order => { if (order.quantity === null) { order.quantity = 1; } }); // 依據餐點關鍵字查詢菜單並建立訂單項目 let orderList = []; dishOrders.forEach(order => { let foundDish = menu.find(item => item.keywords.includes(order.keyword)); if (!foundDish) { alert("未找到與 " + order.keyword + " 相符的餐點！"); return; } // 若餐點價格為 0（免費餐點），依照原邏輯附加到前一筆非免費訂單備註（或獨立列出） if (foundDish.price === 0) { if (orderList.length > 0 && orderList[orderList.length - 1].dish.price !== 0) { let prevRemark = orderList[orderList.length - 1].remark; orderList[orderList.length - 1].remark = prevRemark ? (prevRemark + ", " + foundDish.name + "×" + order.quantity) : (foundDish.name + "×" + order.quantity); } else { orderList.push({ dish: foundDish, quantity: "", remark: foundDish.name + "×" + order.quantity }); } } else { orderList.push({ dish: foundDish, quantity: order.quantity, remark: "" }); } }); if (orderList.length === 0) return; // 組合訂單摘要供使用者確認 let summary = ""; orderList.forEach(order => { if (order.dish.price === 0) { summary += "(" + order.remark + ") " + (order.quantity !== "" ? order.quantity + " 份" : "") + "\n"; } else { summary += order.dish.name + " " + order.quantity + " 份, 單價：" + order.dish.price + " 元\n"; } }); if (confirm("您點的是：\n" + summary + "\n確定要下單嗎？")) { orderList.forEach(order => { addOrderRow(order.dish, order.quantity, order.remark); }); document.getElementById("orderInput").value = ""; } }
